About Data Structures

Data Structures are fundamental concepts in computer science that involve organizing and storing data efficiently. They enable algorithms to process data quickly and effectively.

Sample Code: Linked List

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def append(self, data):
        new_node = Node(data)
        if self.head is None:
            self.head = new_node
            return
        last = self.head
        while last.next:
            last = last.next
        last.next = new_node

    def print_list(self):
        curr_node = self.head
        while curr_node:
            print(curr_node.data)
            curr_node = curr_node.next

# Sample usage
llist = LinkedList()
llist.append(1)
llist.append(2)
llist.append(3)
llist.print_list()
